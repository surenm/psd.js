  constructor: (@file) ->

  parse: ->
    @at = @file.tell()  

    [@type, @id, @namelen] = @file.readf ">4s H B"

    Log.debug "Resource ##{@id}: type=#{@type}"

    n = Util.pad2(@namelen + 1) - 1
    [@name] = @file.readf ">#{n}s"
    @name = @name.substr(0, @name.length - 1)
    @shortName = @name.substr(0, 20)

    @size = @file.readInt()
    @size = Util.pad2(@size)

    if 2000 <= @id <= 2998
      @rdesc = "[Path Information]"
      @file.seek @size
    else if @id is 2999
      assert 0
    else if 4000 <= @id < 5000
      @rdesc = "[Plug-in Resource]"
      @file.seek @size
    else if RESOURCE_DESCRIPTIONS[@id]?
      resource = RESOURCE_DESCRIPTIONS[@id]
      @rdesc = "[#{resource.name}]"

      if resource.parse?
        resource.parse.call(@)
      else
        @file.seek @size
    else
      @file.seek @size

  toJSON: ->
    sections = [
      'type'
      'id'
      'name'
      'rdesc'
    ]

    data = {}
    for section in sections
      data[section] = @[section]

    data
